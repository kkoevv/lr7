cmake_minimum_required(VERSION 3.16)
project(DungeonEditor LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra")

include_directories(include)

add_executable(DungeonEditor
    src/main.cpp
    src/npc.cpp
    src/Bear.cpp
    src/Werewolf.cpp
    src/Rogue.cpp
    src/GameManager.cpp
)

enable_testing()

find_package(GTest REQUIRED)
include_directories(${GTEST_INCLUDE_DIRS})

add_executable(simple_tests
    test/simple_tests.cpp
    src/npc.cpp
    src/Bear.cpp
    src/Werewolf.cpp
    src/Rogue.cpp
    src/GameManager.cpp
)

target_link_libraries(simple_tests ${GTEST_LIBRARIES} pthread)

add_test(NAME simple_tests COMMAND simple_tests)
